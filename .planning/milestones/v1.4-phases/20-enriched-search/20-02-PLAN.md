    2. Add a module-level `isBookingUrl()` helper (not exported):
       ```typescript
       function isBookingUrl(url: string): boolean {
         if (!url) return false;
         try {
           const hostname = new URL(url).hostname.replace(/^www\./, '');
           return BOOKING_DOMAINS.some(
             (domain) => hostname === domain || hostname.endsWith('.' + domain),
           );
         } catch {
           // Fallback for malformed URLs (e.g., Google Maps redirects)
           return BOOKING_DOMAINS.some((domain) => url.includes(domain));
         }
       }
       ```

    3. Add a module-level `formatReviewCount()` helper (not exported):
       ```typescript
       function formatReviewCount(count: number): string {
         return count >= 1000 ? `${(count / 1000).toFixed(1)}K` : String(count);
       }
       ```

    4. Add a module-level `formatOneLiner()` helper (not exported):
       ```typescript
       function formatOneLiner(r: SearchResult, index: number): string {
         let line = `${index + 1}. ${r.title}`;

         if (r.rating !== null && r.rating !== undefined) {
           const reviews =
             r.reviewCount !== null && r.reviewCount !== undefined
               ? ` (${formatReviewCount(r.reviewCount)})`
               : '';
           line += ` ⭐ ${r.rating}${reviews}`;
         }

         if (r.address) {
           line += ` — ${r.address}`;
         }

         if (r.url) {
           const urlPart = isBookingUrl(r.url) ? `🛒 ${r.url}` : r.url;
           line += ` — ${urlPart}`;
         }

         return line;
       }
       ```

    5. Replace the `cards` building block in `formatTravelResults()`:
       - Old code: `const cards = results.map((r, i) => { const lines: string[] = []; ... return lines.join('\n'); });`
       - New code: `const lines = results.map((r, i) => formatOneLiner(r, i));`
       - Change the return from `${header}\n\n${cards.join('\n\n')}` to `${header}\n\n${lines.join('\n')}`
       (One-liners separated by single newlines, not double — keeps the block compact for WhatsApp)

    6. Keep the `formatHelpText()` function entirely unchanged.
