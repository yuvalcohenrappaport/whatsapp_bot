```ts
export interface Coords { lat: number; lon: number }
export interface ForecastSlot { dt: number; temp: number; description: string; icon: string }

export async function resolveCoords(destination: string, apiKey: string): Promise<Coords | null>
// GET https://api.openweathermap.org/geo/1.0/direct?q={destination}&limit=1&appid={apiKey}
// Returns { lat, lon } from first result, or null on empty/error.
// Does NOT cache — the caller (dayOfBriefing) stores coords in metadata.coords.

export async function getDestinationForecast(
  coords: Coords,
  apiKey: string,
  targetDateIso: string, // 'YYYY-MM-DD' — filter slots whose dt falls on this date
): Promise<ForecastSlot[]>
// GET https://api.openweathermap.org/data/2.5/forecast?lat={lat}&lon={lon}&appid={apiKey}&units=metric&cnt=40
// Filter returned `list` items where new Date(slot.dt * 1000).toISOString().startsWith(targetDateIso)
// Returns array of matching slots (typically 2-4 for a given day)
// On HTTP 429: wait exactly 5000ms then retry once. On second failure: throw.
// On any other non-200: throw Error(`OpenWeather ${status}`).
```

```ts
import { describe, it, expect, vi } from 'vitest';
import { resolveDestinationTz, isInBriefingWindow } from '../briefingCron.js';

// Mock config + db — briefingCron imports both
vi.mock('../../config.js', () => ({ config: { LOG_LEVEL: 'silent', GEMINI_API_KEY: 'fake', GEMINI_MODEL: 'gemini-2.5-flash' } }));

  generateText: vi.fn().mockResolvedValue('Europe/Rome'),
}));

describe('resolveDestinationTz', () => {
  it('resolves "rome" → Europe/Rome from table', async () => {
    expect(await resolveDestinationTz('Rome')).toBe('Europe/Rome');
  });
  it('resolves partial match "central rome" → Europe/Rome', async () => {
    expect(await resolveDestinationTz('central Rome')).toBe('Europe/Rome');
  });
  it('uses cached tz when provided', async () => {
    expect(await resolveDestinationTz('Unknown City', 'Pacific/Auckland')).toBe('Pacific/Auckland');
  });
  it('falls back to Gemini for unlisted destination', async () => {
    const tz = await resolveDestinationTz('Reykjavik');
    expect(typeof tz).toBe('string');
    expect(tz.length).toBeGreaterThan(0);
  });
});

describe('isInBriefingWindow', () => {
  // Use fixed nowMs so test is deterministic regardless of wall clock.
  // 2026-05-10T06:02:00Z = 08:02 Europe/Rome (UTC+2 in May, no DST issue)
  const NOW_ROME_0802 = Date.parse('2026-05-10T06:02:00Z');

  it('returns true when time matches within 7min and date is in window', () => {
    expect(isInBriefingWindow({
      nowMs: NOW_ROME_0802,
      destTz: 'Europe/Rome',
      startDate: '2026-05-10',
      endDate: '2026-05-15',
      briefingTime: '08:00',
      lastBriefingDate: null,
    })).toBe(true);
  });
  it('returns false when already briefed today', () => {
    expect(isInBriefingWindow({
      nowMs: NOW_ROME_0802,
      destTz: 'Europe/Rome',
      startDate: '2026-05-10',
      endDate: '2026-05-15',
      briefingTime: '08:00',
      lastBriefingDate: '2026-05-10',
    })).toBe(false);
  });
  it('returns true on day-before-travel (start_date − 1)', () => {
    expect(isInBriefingWindow({
      nowMs: NOW_ROME_0802,
      destTz: 'Europe/Rome',
      startDate: '2026-05-11', // trip starts tomorrow
      endDate: '2026-05-15',
      briefingTime: '08:00',
      lastBriefingDate: null,
    })).toBe(true);
  });
  it('returns false when today is before start_date − 1', () => {
    expect(isInBriefingWindow({
      nowMs: NOW_ROME_0802,
      destTz: 'Europe/Rome',
      startDate: '2026-05-15', // trip starts in 5 days
      endDate: '2026-05-20',
      briefingTime: '08:00',
      lastBriefingDate: null,
    })).toBe(false);
  });
  it('returns false when today is after end_date', () => {
    expect(isInBriefingWindow({
      nowMs: NOW_ROME_0802,
      destTz: 'Europe/Rome',
      startDate: '2026-05-01',
      endDate: '2026-05-09', // ended yesterday
      briefingTime: '08:00',
      lastBriefingDate: null,
    })).toBe(false);
  });
  it('returns false when time is 10min outside briefing_time window', () => {
    // 08:10 Rome — outside ±7min of 08:00

      nowMs: NOW_0810,
      destTz: 'Europe/Rome',
      startDate: '2026-05-10',
      endDate: '2026-05-15',
      briefingTime: '08:00',
      lastBriefingDate: null,

      startDate: '2026-07-01',
      endDate: '2026-07-07',
      briefingTime: '08:00',
      lastBriefingDate: null,
    })).toBe(true);
  });
});
```
