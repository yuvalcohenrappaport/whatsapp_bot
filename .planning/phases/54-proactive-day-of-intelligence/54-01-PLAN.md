export interface ForecastSlot { dt: number; temp: number; description: string; icon: string }

export async function resolveCoords(destination: string, apiKey: string): Promise<Coords | null>

// Returns { lat, lon } from first result, or null on empty/error.

  coords: Coords,

): Promise<ForecastSlot[]>
// GET https://api.openweathermap.org/data/2.5/forecast?lat={lat}&lon={lon}&appid={apiKey}&units=metric&cnt=40
