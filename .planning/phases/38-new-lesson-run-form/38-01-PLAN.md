```python
@router.post("/generate")
async def generate_lesson_run(
    body: GenerateLessonRunRequest,
    request: Request,
    bg: BackgroundTasks,
):
    """Kick off Phase 1 of lesson-mode: generate 4 lesson candidates.
    
    Finds the most recent sequence for the given project_name that has
    context_json populated. Creates a PENDING_LESSON_SELECTION post with
    4 lesson candidates. Returns 202 + job_id.
    """
    with get_conn() as conn:
        row = conn.execute(
            "SELECT id, context_json FROM sequences "
            "WHERE project_name = ? AND context_json IS NOT NULL AND context_json != '' "
            "ORDER BY created_at DESC LIMIT 1",
            (body.project_name,),
        ).fetchone()
    if row is None:
        raise APIError(
            ErrorCode.NOT_FOUND,
            f"no sequence found for project '{body.project_name}' with context",
            {"project_name": body.project_name},
        )

    tracker = get_tracker(request)
    job = await tracker.create("lesson_candidates_generation")
    bg.add_task(
        workers.run_lesson_candidates_generation,
        tracker,
        job.id,
        row["id"],  # source_sequence_id
        body.project_name,
        body.perspective,
        body.language,
        body.topic_hint,
    )
    return JSONResponse(
        status_code=202,
        content=JobAccepted(job_id=job.id).model_dump(mode="json"),
    )
```

```python
async def run_lesson_candidates_generation(
    tracker: "JobTracker",
    job_id: str,
    source_sequence_id: str,
    project_name: str,
    perspective: str,
    language: str,
    topic_hint: Optional[str],
) -> None:
    """Phase 38 — Generate 4 lesson candidates (Phase 1 of lesson mode).
    
    Reuses context_json from the source sequence, generates candidates via
    PostGenerator.generate_lesson_candidates, creates a new PENDING_LESSON_SELECTION
    sequence+post, and persists the candidates.
    """
    def thunk() -> dict:
        from generation.generator import PostGenerator
        from generation.models import Post, PostSequence
        from ingestion.models import ProjectContext
        from review.manager import ReviewManager

        db_path = _db_path_str()
        mgr = ReviewManager(db_path)

        raw = mgr.get_sequence_context_json(source_sequence_id)
        if raw is None:
            raise APIError(ErrorCode.NOT_FOUND, ...)
        if not raw:
            raise APIError(ErrorCode.UNPROCESSABLE, ...)
        
        ctx_dict = json.loads(raw)
        context = ProjectContext(**ctx_dict)

        # If topic_hint provided, inject it into the context description
        # to steer Claude's lesson candidate selection
        if topic_hint:
            context.description = f"{context.description}\n\nTopic focus requested: {topic_hint}"

        generator = PostGenerator()
        candidates = generator.generate_lesson_candidates(
            context,
            perspective=perspective,
            language=language,
        )

        # Create PENDING_LESSON_SELECTION post + sequence (mirrors generate.py --mode lesson)
        placeholder = Post(
            content="(pending lesson selection)",
            status="PENDING_LESSON_SELECTION",
        )
        new_sequence = PostSequence(
            project_name=project_name,
            posts=[placeholder],
            perspective=perspective,
            language=language,
            mode="lesson",
        )
        new_sequence.context_json = raw  # reuse source context_json
        mgr.save_sequence(new_sequence)

        post_id = placeholder.id
        mgr.save_lesson_candidates(post_id, candidates)

        return {
            "sequence_id": new_sequence.id,
            "post_id": post_id,
            "candidate_count": len(candidates),
            "project_name": project_name,
        }

    await _run_with_semaphore(tracker, job_id, thunk)
```

Note: The `_run_with_semaphore` pattern already exists in workers.py and handles the 409 "generator busy" case. Import `Optional` from typing if not already imported.
