Create the new module. It mirrors the structure of `src/commitments/commitmentPipeline.ts::processCommitment` but:
1. Writes a single `actionables` row per extracted item (no dual-path split).
2. Does NOT call Google Tasks.
3. Does NOT send any self-chat notification.
4. Preserves all 4 guards byte-for-byte — reuse the existing `isBlocklisted` / `isIncomingAllowed` / `isOnCooldown` helpers from `commitmentPipeline.ts` by exporting them (add `export` to the existing `function isBlocklisted(...)` etc. in commitmentPipeline.ts so both paths share the same predicates) OR inline them verbatim with a `// MIRRORED FROM commitmentPipeline.ts — keep in sync` comment. Pick whichever is cleaner.

Core shape:

```typescript
import { randomUUID } from 'node:crypto';
import pino from 'pino';
import { config } from '../config.js';
import { getSetting } from '../db/queries/settings.js';
import { commitmentDetection } from '../commitments/CommitmentDetectionService.js';
import { detectMessageLanguage } from '../calendar/calendarApproval.js';
import { createActionable } from '../db/queries/actionables.js';
// Reuse the existing module-local cooldown map + helpers from commitmentPipeline
// OR inline them — see Task decision above.

const logger = pino({ level: config.LOG_LEVEL });

const COOLDOWN_MS = 5 * 60 * 1000;
const chatCooldowns = new Map<string, number>();

function isOnCooldown(chatJid: string): boolean { /* ... */ }
function isBlocklisted(contactJid: string): boolean { /* ... */ }
function isIncomingAllowed(contactJid: string): boolean { /* ... */ }

export async function processDetection(params: {
  messageId: string;
  contactJid: string;
  contactName: string | null;
  text: string;
  timestamp: number;
  fromMe: boolean;
}): Promise<void> {
  try {
    const { messageId, contactJid, contactName, text, timestamp, fromMe } = params;

    // Guards — exact same order + exact same checks as commitmentPipeline.processCommitment
    if (getSetting('commitment_detection_enabled') === 'false') return;
    if (contactJid === config.USER_JID) return;              // self-chat skip
    if (!fromMe && !isIncomingAllowed(contactJid)) return;   // incoming allowlist
    if (isBlocklisted(contactJid)) return;                   // blocklist
    if (!commitmentDetection.passesPreFilter(text, fromMe)) return; // pre-filter
    if (isOnCooldown(contactJid)) return;                    // cooldown
    chatCooldowns.set(contactJid, Date.now());               // set cooldown BEFORE async Gemini call

    // Gemini extraction — reuse the existing service verbatim
    const results = await commitmentDetection.extractCommitments(text, {
      contactName,
      contactJid,
      fromMe,
    });
    if (results.length === 0) return;

    // Write one actionable per extracted item — NO Google Tasks call, NO self-chat message
    const detectedLanguage = detectMessageLanguage(text);
    for (const item of results) {
      const id = randomUUID();
      createActionable({
        id,
        sourceType: item.type, // 'commitment' | 'task' — persisted verbatim
        sourceContactJid: contactJid,
        sourceContactName: contactName,
        sourceMessageId: messageId,
        sourceMessageText: text,
        detectedLanguage,
        originalDetectedTask: item.task,
        task: item.task,
        status: 'pending_approval',
        detectedAt: timestamp,
        fireAt: item.dateTime?.getTime() ?? null, // null for timeless tasks
      });
      logger.info({ id, task: item.task, sourceType: item.type, contactJid }, 'Actionable detected (pending_approval)');
    }
  } catch (err) {
    logger.error({ err, messageId: params.messageId }, 'Error in detection pipeline — skipping');
  }
}
```

Key invariants for the planner/executor:
- **No Google Tasks calls** — grep the final file for `createTodoTask`, `todoService`, `todoPipeline`, `processDetectedTask` must ALL return zero matches.
- **No self-chat messages** — grep for `sendMessage`, `getState`, `USER_JID` must show only the self-chat-skip guard, nothing else.
- **Guards in the exact order above** — the order matters for cost semantics (cheap checks before Gemini call).
- **`chatCooldowns.set(contactJid, Date.now())` happens BEFORE the `await extractCommitments`** — this is the race-condition fix from the existing pipeline (comment references "avoid race condition"). Preserve it.
