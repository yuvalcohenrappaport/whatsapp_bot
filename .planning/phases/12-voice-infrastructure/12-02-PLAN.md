<task type="auto">

  <action>
    **Create src/voice/client.ts** (new file, new directory):

    ```typescript
    import { ElevenLabsClient } from '@elevenlabs/elevenlabs-js';
    import ffmpegPath from 'ffmpeg-static';
    import pino from 'pino';
    import { config } from '../config.js';

    export const elevenLabsClient = new ElevenLabsClient({
      apiKey: config.ELEVENLABS_API_KEY,
    });

    /**
     * Validates that the ElevenLabs API key and configured voice ID are valid.
     * Non-blocking: logs a warning on failure but returns false instead of throwing.
     * Uses a 5s timeout and 0 retries to avoid hanging startup.
     */
    export async function validateElevenLabsConnection(logger: pino.Logger): Promise<boolean> {
      // Verify ffmpeg-static binary is available
      if (!ffmpegPath) {
        logger.warn('ffmpeg-static binary not found — audio encoding will fail in voice pipeline');
      } else {
        logger.info({ ffmpegPath }, 'ffmpeg-static binary resolved');
      }

      // Validate ElevenLabs API key + voice ID
      try {
        await elevenLabsClient.voices.get(config.ELEVENLABS_DEFAULT_VOICE_ID, {
          timeoutInSeconds: 5,
          maxRetries: 0,
        } as Parameters<typeof elevenLabsClient.voices.get>[1]);
        logger.info('ElevenLabs connection validated — voice replies available');
        return true;
      } catch (err) {
        logger.warn({ err }, 'ElevenLabs validation failed — voice replies will be unavailable until config is fixed');
        return false;
      }
    }
    ```

    Notes on this implementation:
    - `import ffmpegPath from 'ffmpeg-static'` is the correct default import (NOT `import * as`). If TypeScript raises a type error, the fix is `npm install -D @types/ffmpeg-static` (per Plan 01 task).
    - The validator is non-blocking: it returns a boolean, does not throw. The caller decides what to do with the result.
    - The `{ timeoutInSeconds: 5, maxRetries: 0 }` options prevent the default 60s × 2 retries = 120s hang on startup if ElevenLabs is unreachable.
    - If the ElevenLabs SDK types don't accept options as the second arg to `voices.get()`, check the SDK's type signature and adjust accordingly (some SDK versions accept options differently — use whatever the SDK type requires).
    - The `ffmpegPath` check uses `ffmpeg-static`'s `string | null` return — the null case is a valid runtime state that must be handled.
