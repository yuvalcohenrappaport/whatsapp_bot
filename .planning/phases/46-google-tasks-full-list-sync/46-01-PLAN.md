### Color hash helper

```ts
const PALETTE = [
  'bg-emerald-500', 'bg-sky-500', 'bg-violet-500',
  'bg-amber-500',   'bg-rose-500', 'bg-teal-500',
  'bg-orange-500',  'bg-fuchsia-500',
];

function hashListColor(listId: string): string {
  // djb2 hash over listId chars
  let h = 5381;
  for (let i = 0; i < listId.length; i++) h = ((h << 5) + h) ^ listId.charCodeAt(i);
  return PALETTE[Math.abs(h) % PALETTE.length];
}
```

(listId is stable per Google; renaming a list does NOT change its id, so color is stable across renames — per CONTEXT locked decision.)

**Add to `src/db/queries/actionables.ts`:**
```ts
export function getApprovedActionableTodoTaskIds(fromMs: number, toMs: number): Set<string> {
  // Returns todoTaskId values for live (approved status) actionables with fireAt in window.
  // Used by the gtasks aggregator to dedup items already mirrored.
  const rows = db.select({ todoTaskId: actionables.todoTaskId })
    .from(actionables)
    .where(
      and(
        eq(actionables.status, 'approved'),
        isNotNull(actionables.todoTaskId),
        gte(actionables.fireAt, fromMs),
        lte(actionables.fireAt, toMs),
      )
    )
    .all();
  return new Set(rows.map(r => r.todoTaskId!));
}
```
