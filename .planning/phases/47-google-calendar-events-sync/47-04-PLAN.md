```bash
  | python3 -c "import sys,json; d=json.load(sys.stdin); print('items:', len(d.get('items',[])), 'sample:', (d.get('items',[{}])[0].get('title','<empty>') if d.get('items') else '<empty>'))"

# Unified aggregator shows gcal
curl -s -H "Authorization: Bearer $JWT" http://localhost:3000/api/calendar/items \
  | python3 -c "import sys,json; d=json.load(sys.stdin); print('sources:', d['sources']); print('gcal items:', len([i for i in d['items'] if i['source']=='gcal']))"

# No-auth → 401
curl -s -o /dev/null -w '%{http_code}\n' http://localhost:3000/api/google-calendar/calendars
curl -s -o /dev/null -w '%{http_code}\n' http://localhost:3000/api/google-calendar/events
```
