```python
"""Canonical builder for PostDTO from the pm-authority state.db schema.

Schema facts (from pm-authority/auth/token_store.py + review/manager.py):
- posts.id is TEXT (UUID), NOT int
- posts has no created_at / updated_at columns; we map created_at from sequences.created_at
- post_variants.angle_type is the 'kind' on the wire
- post_variants.selected: 0=unselected, 1=chosen, -1=superseded (exclude from DTO array)
- lesson_candidates has lesson_text, rationale, selected
- regeneration_capped := regeneration_count >= 5
- Position := 1-based index of this post's id within the sequence's post list (ordered by rowid)
- pii_reviewed := post.status != 'PENDING_PII_REVIEW' (derived; CONTEXT.md §2 wire shape)
"""
from __future__ import annotations
import sqlite3
from datetime import datetime, timezone
from typing import Iterable, Optional

from services.http.errors import APIError, ErrorCode
from services.http.schemas import (
    PostDTO, VariantDTO, LessonCandidateDTO, ImageInfoDTO,
)

NON_TERMINAL_STATUSES = (
    "DRAFT", "PENDING_REVIEW", "PENDING_VARIANT",
    "PENDING_LESSON_SELECTION", "PENDING_PII_REVIEW", "APPROVED",
)

def _parse_ts(raw: Optional[str]) -> Optional[datetime]:
    if not raw:
        return None
    try:
        dt = datetime.fromisoformat(raw)
    except ValueError:
        return None
    if dt.tzinfo is None:
        dt = dt.replace(tzinfo=timezone.utc)
    return dt.astimezone(timezone.utc)

def _variant_rows_to_dtos(rows: Iterable[sqlite3.Row]) -> list[VariantDTO]:
    out: list[VariantDTO] = []
    for r in rows:
        if r["selected"] == -1:
            continue  # superseded — excluded from wire
        out.append(VariantDTO(
            id=r["id"],
            kind=r["angle_type"],
            content=r["content"],
            image_prompt=r["image_prompt"],
            selected=bool(r["selected"]),
        ))
    return out

def _lesson_rows_to_dtos(post_id: str, rows: Iterable[sqlite3.Row]) -> list[LessonCandidateDTO]:
    out: list[LessonCandidateDTO] = []
    for r in rows:
        out.append(LessonCandidateDTO(
            id=r["id"],
            lesson_text=r["lesson_text"],
            rationale=r["rationale"],
            image_url=f"/v1/posts/{post_id}/lesson-candidates/{r['id']}/image",
            selected=bool(r["selected"]),
        ))
    return out

def _image_info(post_row: sqlite3.Row) -> ImageInfoDTO:
    image_path = post_row["image_path"]
    source = post_row["image_source"]
    # pii_reviewed is derived: anything past the PII review gate is considered reviewed.
    # Only posts currently sitting in PENDING_PII_REVIEW are NOT yet reviewed.
    pii_reviewed = post_row["status"] != "PENDING_PII_REVIEW"
    if image_path:
        return ImageInfoDTO(
            source=source if source in ("ai", "screenshot") else None,
            url=f"/v1/posts/{post_row['id']}/image",
            pii_reviewed=pii_reviewed,
            filesystem_path=image_path,  # Field(exclude=True) — never on wire
        )
    return ImageInfoDTO(source=None, url=None, pii_reviewed=pii_reviewed, filesystem_path=None)

def _position_in_sequence(conn: sqlite3.Connection, sequence_id: str, post_id: str) -> int:
    rows = conn.execute(
        "SELECT id FROM posts WHERE sequence_id = ? ORDER BY rowid", (sequence_id,)
    ).fetchall()
    for idx, r in enumerate(rows, start=1):
        if r["id"] == post_id:
            return idx
    return 0

def build_post_dto(conn: sqlite3.Connection, post_id: str) -> PostDTO:
    post = conn.execute(
        """SELECT id, sequence_id, content, content_he, status, image_path,
                  image_source, share_urn, scheduled_at, published_at,
                  regeneration_count
           FROM posts WHERE id = ?""",
        (post_id,),
    ).fetchone()
    if not post:
        raise APIError(ErrorCode.NOT_FOUND, f"post {post_id} not found", {"post_id": post_id})

    seq = conn.execute(
        "SELECT id, created_at, perspective, language FROM sequences WHERE id = ?",
        (post["sequence_id"],),
    ).fetchone()
    if not seq:
        raise APIError(
            ErrorCode.INTERNAL_ERROR,
            "post references missing sequence",
            {"post_id": post_id, "sequence_id": post["sequence_id"]},
        )

    variant_rows = conn.execute(
        "SELECT id, angle_type, content, image_prompt, selected FROM post_variants WHERE post_id = ? ORDER BY id",
        (post_id,),
    ).fetchall()
    lesson_rows = conn.execute(
        "SELECT id, lesson_text, rationale, selected FROM lesson_candidates WHERE post_id = ? ORDER BY id",
        (post_id,),
    ).fetchall()

    regen_count = int(post["regeneration_count"] or 0)

    return PostDTO(
        id=post["id"],
        sequence_id=post["sequence_id"],
        position=_position_in_sequence(conn, post["sequence_id"], post_id),
        status=post["status"],
        perspective=seq["perspective"],
        language=seq["language"],
        content=post["content"],
        content_he=post["content_he"],
        image=_image_info(post),
        variants=_variant_rows_to_dtos(variant_rows),
        lesson_candidates=_lesson_rows_to_dtos(post_id, lesson_rows),
        regeneration_count=regen_count,
        regeneration_capped=regen_count >= 5,
        share_urn=post["share_urn"],
        scheduled_at=_parse_ts(post["scheduled_at"]),
        published_at=_parse_ts(post["published_at"]),
        created_at=_parse_ts(seq["created_at"]) or datetime.now(timezone.utc),
        updated_at=None,
    )

def list_post_dtos(conn: sqlite3.Connection, statuses: list[str] | None) -> list[PostDTO]:
    if statuses:
        placeholders = ",".join("?" * len(statuses))
        ids_rows = conn.execute(
            f"SELECT id FROM posts WHERE status IN ({placeholders}) ORDER BY rowid DESC",
            statuses,
        ).fetchall()
    else:
        placeholders = ",".join("?" * len(NON_TERMINAL_STATUSES))
        ids_rows = conn.execute(
            f"SELECT id FROM posts WHERE status IN ({placeholders}) ORDER BY rowid DESC",
            list(NON_TERMINAL_STATUSES),
        ).fetchall()
    return [build_post_dto(conn, r["id"]) for r in ids_rows]
```

```python
# ---- Routes appended by Plan 33-02 ----

from pathlib import Path
import mimetypes
from fastapi.responses import FileResponse
from services.http.db import get_conn
from services.http.errors import APIError, ErrorCode

PM_ROOT = Path("/home/yuval/pm-authority").resolve()
DATA_ROOT = (PM_ROOT / "data").resolve()

def _safe_image_response(raw_path: str | None, identifier: dict) -> FileResponse:
    if not raw_path:
        raise APIError(ErrorCode.NOT_FOUND, "no image for this resource", identifier)
    try:
        resolved = Path(raw_path).resolve()
    except Exception:
        raise APIError(ErrorCode.NOT_FOUND, "invalid image path", identifier)
    try:
        resolved.relative_to(DATA_ROOT)
    except ValueError:
        raise APIError(
            ErrorCode.NOT_FOUND,
            "image path outside data root",
            {**identifier, "path": str(resolved)},
        )
    if not resolved.is_file():
        raise APIError(ErrorCode.NOT_FOUND, "image file missing on disk", identifier)
    media_type, _ = mimetypes.guess_type(resolved.name)
    return FileResponse(str(resolved), media_type=media_type or "application/octet-stream")

@router.get("/{post_id}/image")
async def get_post_image(post_id: str):
    with get_conn() as conn:
        row = conn.execute(
            "SELECT image_path FROM posts WHERE id = ?", (post_id,)
        ).fetchone()
    if not row:
        raise APIError(ErrorCode.NOT_FOUND, "post not found", {"post_id": post_id})
    return _safe_image_response(row["image_path"], {"post_id": post_id})

@router.get("/{post_id}/lesson-candidates/{candidate_id}/image")
async def get_lesson_candidate_image(post_id: str, candidate_id: int):
    with get_conn() as conn:
        row = conn.execute(
            "SELECT selected FROM lesson_candidates WHERE id = ? AND post_id = ?",
            (candidate_id, post_id),
        ).fetchone()
        if not row:
            raise APIError(
                ErrorCode.NOT_FOUND,
                "lesson candidate not found",
                {"post_id": post_id, "candidate_id": candidate_id},
            )
        if not row["selected"]:
            raise APIError(
                ErrorCode.NOT_FOUND,
                "lesson candidate has no image yet (not selected or image not generated)",
                {"post_id": post_id, "candidate_id": candidate_id},
            )
        post_row = conn.execute(
            "SELECT image_path FROM posts WHERE id = ?", (post_id,)
        ).fetchone()
    return _safe_image_response(
        post_row["image_path"] if post_row else None,
        {"post_id": post_id, "candidate_id": candidate_id},
    )
```
