from enum import Enum
from typing import Any
from pydantic import BaseModel

class ErrorCode(str, Enum):
    VALIDATION_ERROR = "VALIDATION_ERROR"        # 400
    NOT_FOUND = "NOT_FOUND"                      # 404
    STATE_VIOLATION = "STATE_VIOLATION"          # 409
    REGEN_CAPPED = "REGEN_CAPPED"                # 409
    LESSON_ALREADY_PICKED = "LESSON_ALREADY_PICKED"  # 409
    VARIANT_ALREADY_PICKED = "VARIANT_ALREADY_PICKED"  # 409
    UNPROCESSABLE = "UNPROCESSABLE"              # 422
    INTERNAL_ERROR = "INTERNAL_ERROR"            # 500
    UPSTREAM_FAILURE = "UPSTREAM_FAILURE"        # 502
    UNAVAILABLE = "UNAVAILABLE"                  # 503

_STATUS_MAP = {
    ErrorCode.VALIDATION_ERROR: 400,
    ErrorCode.NOT_FOUND: 404,
    ErrorCode.STATE_VIOLATION: 409,
    ErrorCode.REGEN_CAPPED: 409,
    ErrorCode.LESSON_ALREADY_PICKED: 409,
    ErrorCode.VARIANT_ALREADY_PICKED: 409,
    ErrorCode.UNPROCESSABLE: 422,
    ErrorCode.INTERNAL_ERROR: 500,
    ErrorCode.UPSTREAM_FAILURE: 502,
    ErrorCode.UNAVAILABLE: 503,
}

def status_for(code: ErrorCode) -> int:
    return _STATUS_MAP[code]

class ErrorDetail(BaseModel):
    code: ErrorCode
    message: str
    details: dict[str, Any] = {}

class ErrorEnvelope(BaseModel):
    error: ErrorDetail

class APIError(Exception):
    """Raise this from routers to surface a structured error."""
    def __init__(self, code: ErrorCode, message: str, details: dict[str, Any] | None = None):
        self.code = code
        self.message = message
        self.details = details or {}
        super().__init__(message)

```python
import sqlite3
import time
import logging
from pathlib import Path
from contextlib import contextmanager

PM_AUTHORITY_ROOT = Path(__file__).resolve().parents[2]
DEFAULT_DB_PATH = PM_AUTHORITY_ROOT / "state.db"

log = logging.getLogger("pm_http.db")

def _apply_pragmas(conn: sqlite3.Connection) -> None:
    conn.execute("PRAGMA journal_mode=WAL")
    conn.execute("PRAGMA busy_timeout=5000")
    conn.execute("PRAGMA foreign_keys=ON")

def open_db_with_retry(db_path: Path = DEFAULT_DB_PATH, *, attempts: int = 3, backoff_seconds: float = 1.666) -> bool:
    """Try to open state.db attempts times over ~5s. Returns True on success."""
    last_err: Exception | None = None
    for i in range(attempts):
        try:
            conn = sqlite3.connect(str(db_path), timeout=5.0)
            try:
                _apply_pragmas(conn)
                conn.execute("SELECT 1 FROM sqlite_master LIMIT 1").fetchone()
            finally:
                conn.close()
            return True
        except sqlite3.Error as e:
            last_err = e
            log.warning("state.db open attempt %d/%d failed: %s", i + 1, attempts, e)
            if i < attempts - 1:
                time.sleep(backoff_seconds)
    log.error("state.db open failed after %d attempts: %s", attempts, last_err)
    return False

@contextmanager
def get_conn(db_path: Path = DEFAULT_DB_PATH):
    conn = sqlite3.connect(str(db_path), timeout=5.0)
    conn.row_factory = sqlite3.Row
    try:
        _apply_pragmas(conn)
        yield conn
    finally:
        conn.close()
```

from fastapi import APIRouter, Request
from services.http.schemas import HealthResponse
from services.http.errors import APIError, ErrorCode

router = APIRouter(tags=["health"])

VERSION = "0.1.0"

@router.get("/health", response_model=HealthResponse)
async def health(request: Request) -> HealthResponse:
    db_ready: bool = getattr(request.app.state, "db_ready", False)
    if not db_ready:
        raise APIError(
            ErrorCode.UNAVAILABLE,
            "state.db is not yet available",
            {"reason": "startup_retry_exhausted_or_in_progress"},
        )
    return HealthResponse(status="ok", version=VERSION, db_ready=True)

```python
"""Jobs router + background GC task registration.

Plan 33-03 appends GET /jobs/{job_id} to `router` and the JobTracker class
that the GC loop drives. Main.py wires `start_gc_task` / `stop_gc_task` as
startup/shutdown event handlers so Plan 33-03 never has to touch main.py.
"""
from fastapi import APIRouter, FastAPI

router = APIRouter(prefix="/jobs", tags=["jobs"])

async def start_gc_task(app: FastAPI) -> None:
    """Startup hook: creates the JobTracker and starts its GC loop.

    In Plan 33-01 this is a no-op placeholder — Plan 33-03 replaces the body
    to instantiate JobTracker and call .start_gc(). main.py always calls this
    function on startup, so the signature is stable.
    """
    tracker = getattr(app.state, "job_tracker", None)
    if tracker is None:
        # Plan 33-01 baseline: no tracker yet. Plan 33-03 fills this in.
        return
    if hasattr(tracker, "start_gc"):
        tracker.start_gc()

async def stop_gc_task(app: FastAPI) -> None:
    """Shutdown hook: stops the JobTracker GC loop cleanly."""
    tracker = getattr(app.state, "job_tracker", None)
    if tracker is not None and hasattr(tracker, "stop_gc"):
        await tracker.stop_gc()
```

```python
import logging
from contextlib import asynccontextmanager
from fastapi import FastAPI
from fastapi.responses import JSONResponse
from fastapi.exceptions import RequestValidationError
from starlette.exceptions import HTTPException as StarletteHTTPException

from services.http.db import open_db_with_retry
from services.http.errors import (
    APIError, ErrorCode, ErrorDetail, ErrorEnvelope, status_for,
)
from services.http.routers import health as health_router
from services.http.routers import posts as posts_router
from services.http.routers import images as images_router
from services.http.routers import jobs as jobs_router
from services.http.routers import mutations_fast as fast_router
from services.http.routers import mutations_slow as slow_router
from services.http.routers import lesson_runs as lesson_runs_router

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s %(levelname)s %(name)s: %(message)s",
)
log = logging.getLogger("pm_http")

@asynccontextmanager
async def lifespan(app: FastAPI):
    # 1. Open state.db with retry (3x over 5s)
    app.state.db_ready = open_db_with_retry()
    if not app.state.db_ready:
        log.error("state.db unavailable — /v1/health will return 503 UNAVAILABLE")
    else:
        log.info("state.db opened successfully")
    # 2. Start job tracker GC (shim — Plan 33-03 provides the real JobTracker)
    await jobs_router.start_gc_task(app)
    yield
    # 3. Stop job tracker GC
    await jobs_router.stop_gc_task(app)

app = FastAPI(
    title="pm-authority HTTP",
    version="0.1.0",
    lifespan=lifespan,
    docs_url=None,
    redoc_url=None,
    openapi_url=None,
)

# ---- Error handlers ----

def _envelope(code: ErrorCode, message: str, details: dict | None = None) -> JSONResponse:
    body = ErrorEnvelope(error=ErrorDetail(code=code, message=message, details=details or {}))
    return JSONResponse(status_code=status_for(code), content=body.model_dump(mode="json"))

@app.exception_handler(APIError)
async def _handle_api_error(_request, exc: APIError):
    return _envelope(exc.code, exc.message, exc.details)

@app.exception_handler(RequestValidationError)
async def _handle_validation(_request, exc: RequestValidationError):
    return _envelope(
        ErrorCode.VALIDATION_ERROR,
        "request body failed validation",
        {"errors": exc.errors()},
    )

@app.exception_handler(StarletteHTTPException)
async def _handle_http_exc(_request, exc: StarletteHTTPException):
    if exc.status_code == 404:
        return _envelope(ErrorCode.NOT_FOUND, str(exc.detail or "not found"))
    if exc.status_code == 405:
        return _envelope(ErrorCode.VALIDATION_ERROR, "method not allowed")
    return _envelope(ErrorCode.INTERNAL_ERROR, "http error", {"status": exc.status_code})

@app.exception_handler(Exception)
async def _handle_unhandled(_request, exc: Exception):
    log.exception("unhandled exception in request", exc_info=exc)
    return _envelope(ErrorCode.INTERNAL_ERROR, "internal error")

# ---- Router includes (all six + health, pre-wired in Plan 33-01) ----
#
# IMPORTANT: Plans 33-02 / 33-03 / 33-04 MUST NOT edit this file. They only
# append route handlers to their respective router modules. The router
# objects themselves already exist and are already mounted here. This is
# the architectural boundary that keeps waves 2-4 file-disjoint and
# parallel-safe.

app.include_router(health_router.router, prefix="/v1")
app.include_router(posts_router.router, prefix="/v1")
app.include_router(images_router.router, prefix="/v1")
app.include_router(jobs_router.router, prefix="/v1")
app.include_router(fast_router.router, prefix="/v1")
app.include_router(slow_router.router, prefix="/v1")
app.include_router(lesson_runs_router.router, prefix="/v1")
```

Note: FastAPI resolves routes by (path, method). `posts_router`, `images_router`, `fast_router`, and `slow_router` all share the `/v1/posts` prefix but operate on different HTTP methods or different sub-paths, so there are no collisions.
