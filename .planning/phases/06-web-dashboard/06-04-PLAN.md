**dashboard/src/hooks/useDrafts.ts:**

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { apiFetch } from '@/api/client';

export interface Draft {
  id: string;
  contactJid: string;
  contactName: string | null;
  inboundMessage: string;    // body of the message the bot is replying to
  body: string;              // the draft reply text
  createdAt: number;
}

export function useDrafts() {
  return useQuery({
    queryKey: ['drafts'],
    queryFn: () => apiFetch<Draft[]>('/api/drafts'),
    refetchInterval: 15_000,  // poll every 15s as backup to manual invalidation
  });
}

export function useApproveDraft() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: ({ id, body }: { id: string; body: string }) =>
      apiFetch(`/api/drafts/${id}/approve`, {
        method: 'PATCH',
        body: JSON.stringify({ body }),
      }),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ['drafts'] });
      // toast is called from the component, not here
    },
  });
}

export function useRejectDraft() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: (id: string) =>
      apiFetch(`/api/drafts/${id}`, { method: 'DELETE' }),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ['drafts'] });
    },
  });
}
```

**dashboard/src/components/drafts/DraftRow.tsx:**

Renders one draft in the approval queue. Layout: horizontal row inside a Card.
- Left column (flex-1): contact name (bold), inbound message snippet (muted, smaller), then the draft reply text or textarea
- Right column: Approve button (green/primary) and Reject button (outline/destructive variant)

Inline edit pattern per user decisions:
- Draft reply text is displayed as a clickable `<p>` — clicking it switches to `<TextareaAutosize>`
- `TextareaAutosize` is auto-focused, minRows=2
- Editing state is local: `const [editing, setEditing] = useState(false)` and `const [body, setBody] = useState(draft.body)`
- Clicking outside (onBlur) exits editing mode without saving (just keeps the edited body in state for when Approve is clicked)
- Approve button: calls `useApproveDraft().mutate({ id: draft.id, body })`, then on success: `toast.success('Sent!')`. Button disabled while mutation is pending.
- Reject button: calls `useRejectDraft().mutate(draft.id)`. No confirmation dialog needed.

```typescript
import TextareaAutosize from 'react-textarea-autosize';
import { toast } from 'sonner';
// ... shadcn Button, Card imports

export function DraftRow({ draft }: { draft: Draft }) {
  const [editing, setEditing] = useState(false);
  const [body, setBody] = useState(draft.body);
  const approve = useApproveDraft();
  const reject = useRejectDraft();

  return (
    <Card className="p-6">
      <div className="flex items-start gap-6">
        <div className="flex-1 space-y-3 min-w-0">
          <div className="font-medium">{draft.contactName ?? draft.contactJid}</div>
          <p className="text-sm text-muted-foreground truncate">{draft.inboundMessage}</p>
          {editing ? (
            <TextareaAutosize
              className="w-full resize-none bg-muted rounded-md p-3 text-sm focus:outline-none focus:ring-2 focus:ring-ring"
              value={body}
              onChange={(e) => setBody(e.target.value)}
              onBlur={() => setEditing(false)}
              autoFocus
              minRows={2}
            />
          ) : (
            <p
              className="text-sm cursor-pointer hover:bg-muted rounded-md p-3 transition-colors"
              onClick={() => setEditing(true)}
              title="Click to edit"
            >
              {body}
            </p>
          )}
        </div>
        <div className="flex flex-col gap-2 shrink-0">
          <Button
            size="sm"
            onClick={() => {
              approve.mutate(
                { id: draft.id, body },
                { onSuccess: () => toast.success('Sent!') },
              );
            }}
            disabled={approve.isPending}
          >
            Approve
          </Button>
          <Button
            size="sm"
            variant="outline"
            onClick={() => reject.mutate(draft.id)}
            disabled={reject.isPending}
          >
            Reject
          </Button>
        </div>
      </div>
    </Card>
  );
}
```

**dashboard/src/hooks/useGroups.ts:**

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { apiFetch } from '@/api/client';

export interface Group {
  id: string;             // WhatsApp group JID
  name: string | null;
  active: boolean;
  reminderDay: string | null;  // 'monday' | 'tuesday' | etc.
  calendarLink: string | null;
  memberEmails: string | null; // JSON array string e.g. '["a@b.com","c@d.com"]'
  createdAt: number;
  updatedAt: number;
}

export function useGroups() {
  return useQuery({ queryKey: ['groups'], queryFn: () => apiFetch<Group[]>('/api/groups') });
}

export function useAddGroup() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: (data: { id: string; name?: string }) =>
      apiFetch('/api/groups', { method: 'POST', body: JSON.stringify(data) }),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['groups'] }),
  });
}

export function useUpdateGroup() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: ({ id, patch }: { id: string; patch: Partial<Omit<Group, 'id' | 'createdAt' | 'updatedAt'>> }) =>
      apiFetch(`/api/groups/${encodeURIComponent(id)}`, {
        method: 'PATCH',
        body: JSON.stringify(patch),
      }),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['groups'] }),
  });
}

export function useDeleteGroup() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: (id: string) =>
      apiFetch(`/api/groups/${encodeURIComponent(id)}`, { method: 'DELETE' }),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['groups'] }),
  });
}
```

**dashboard/src/components/status/QRModal.tsx:**

```typescript
import { useEffect } from 'react';
import { QRCodeSVG } from 'qrcode.react';
import {
  Dialog, DialogContent, DialogHeader, DialogTitle, DialogDescription,
} from '@/components/ui/dialog';
import type { ConnectionStatus } from '@/hooks/useConnectionStatus';

interface QRModalProps {
  open: boolean;
  qr: string | null;
  status: ConnectionStatus;
  onClose: () => void;
}

export function QRModal({ open, qr, status, onClose }: QRModalProps) {
  // Auto-close when connection becomes established
  useEffect(() => {
    if (status === 'connected' && open) {
      onClose();
    }
  }, [status, open, onClose]);

  return (
    <Dialog open={open} onOpenChange={(isOpen) => { if (!isOpen) onClose(); }}>
      <DialogContent className="sm:max-w-sm">
        <DialogHeader>
          <DialogTitle>Scan QR Code</DialogTitle>
          <DialogDescription>
            Open WhatsApp on your phone, go to Linked Devices, and scan this code.
          </DialogDescription>
        </DialogHeader>
        <div className="flex items-center justify-center p-6">
          {qr ? (
            <QRCodeSVG value={qr} size={240} bgColor="#ffffff" fgColor="#000000" />
          ) : (
            <p className="text-muted-foreground text-sm">
              Waiting for QR code... Make sure the bot is running.
            </p>
          )}
        </div>
      </DialogContent>
    </Dialog>
  );
}
```

```typescript
interface TopbarProps {

  qr: string | null;

export function Topbar({ status, qr }: TopbarProps) {
  const [qrModalOpen, setQrModalOpen] = useState(false);

  return (
    <>

        status={status}
```
