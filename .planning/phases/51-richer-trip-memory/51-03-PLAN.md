<objective>
Ship three self-report slash commands — `!pref`, `!budget`, `!dates` — that let group members explicitly record preferences, per-category budgets, and trip dates without relying on the classifier's inference. Malformed commands are silently discarded (discreet chattiness).

Purpose: MEM2-04 requires `self_reported` as an origin label distinct from `inferred`. MEM2-02 needs a user path to seed `budget_by_category`. Without explicit commands, users would need the classifier to catch everything — unreliable for setup data like dates and budgets.

Output: new `src/groups/tripPreferences.ts` handler, updated pipeline, unit tests.
</objective>

    ```ts
    import crypto from 'node:crypto';
    import pino from 'pino';
    import { config } from '../config.js';
    import { insertTripDecision, upsertTripContext, getTripContext } from '../db/queries/tripMemory.js';

    const logger = pino({ level: config.LOG_LEVEL });

    const VALID_CATEGORIES = ['flights','lodging','food','activities','transit','shopping','other'] as const;
    type Category = typeof VALID_CATEGORIES[number];

    const ISO_DATE_RE = /^\d{4}-\d{2}-\d{2}$/;
    const CURRENCY_RE = /^[A-Z]{3}$/;

    interface GroupMsg {
      id: string;
      senderJid: string;
      senderName: string | null;
      body: string;
      timestamp: number;
    }

    /**
     * Returns true if the message matched a known self-report verb (handled, terminal).
     * Returns false if the message is not a `!`-prefixed command OR uses an unknown verb (pass through).
     * Malformed but-known-verb commands are silently swallowed — return true (terminal), no DB write, no reply.
     */
    export async function handleSelfReportCommand(groupJid: string, msg: GroupMsg): Promise<boolean> {
      const trimmed = msg.body.trim();
      if (!trimmed.startsWith('!')) return false;

      const [verbRaw, ...rest] = trimmed.slice(1).split(/\s+/);
      const verb = verbRaw?.toLowerCase();

      if (verb === 'pref') {
        return handlePref(groupJid, msg, rest.join(' '));
      }
      if (verb === 'budget') {
        return handleBudget(groupJid, msg, rest);
      }
      if (verb === 'dates') {
        return handleDates(groupJid, msg, rest);
      }
      return false; // unknown verb — let it fall through to classifier (user might just be shouting)
    }

    function handlePref(groupJid: string, msg: GroupMsg, text: string): true {
      if (text.trim().length === 0) return true; // malformed, silent
      insertTripDecision({
        id: crypto.randomUUID(),
        groupJid,
        type: 'activity', // best-fit; the `value` text carries the nuance
        value: text.trim(),
        confidence: 'high',
        sourceMessageId: msg.id,
        proposedBy: msg.senderJid,
        origin: 'self_reported',
        category: null,
        costAmount: null,
        costCurrency: null,
      });
      logger.info({ groupJid, senderJid: msg.senderJid }, '!pref recorded');
      return true;
    }

    function handleBudget(groupJid: string, msg: GroupMsg, args: string[]): true {
      if (args.length !== 3) return true; // malformed, silent
      const [categoryRaw, amountRaw, currencyRaw] = args;
      const category = categoryRaw?.toLowerCase();
      const amount = Number(amountRaw);
      const currency = currencyRaw?.toUpperCase();

      if (!VALID_CATEGORIES.includes(category as Category)) return true;
      if (!Number.isFinite(amount) || amount <= 0) return true;
      if (!currency || !CURRENCY_RE.test(currency)) return true;

      // Read current budget, merge, write back
      const ctx = getTripContext(groupJid);
      const current: Record<string, { amount: number; currency: string }> =
        ctx?.budgetByCategory ? JSON.parse(ctx.budgetByCategory) : {};
      current[category] = { amount, currency };

      upsertTripContext(groupJid, {
        budgetByCategory: current, // upsertTripContext stringifies
      });
      logger.info({ groupJid, category, amount, currency }, '!budget recorded');
      return true;
    }

    function handleDates(groupJid: string, msg: GroupMsg, args: string[]): true {
      if (args.length !== 2) return true; // malformed, silent
      const [startDate, endDate] = args;
      if (!ISO_DATE_RE.test(startDate) || !ISO_DATE_RE.test(endDate)) return true;
      if (startDate > endDate) return true; // silent reject

      upsertTripContext(groupJid, { startDate, endDate });
      logger.info({ groupJid, startDate, endDate }, '!dates recorded');
      return true;
    }
    ```

    **Notes:**
    - `getTripContext` currently returns the row with the new columns if 51-01 landed — if `budgetByCategory` field doesn't exist on the returned row (schema mismatch), JSON.parse will throw — wrap in try/catch returning {}.
    - `upsertTripContext` in 51-01 accepts `budgetByCategory` as an object and serializes internally. If that wasn't implemented in 51-01 exactly as specified, this plan's task must inline the JSON.stringify. Verify against 51-01-SUMMARY during execution.
    - `handlePref` uses `type='activity'` because the existing `tripDecisions.type` enum doesn't have a `preference` value. Acceptable — the `origin='self_reported'` field distinguishes preferences from classifier-inferred activities. Dashboard (Phase 55) will group by origin.
