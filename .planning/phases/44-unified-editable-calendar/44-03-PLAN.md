<objective>
Ship the unified calendar read + SSE surface that the dashboard calendar page will consume — PLUS three per-source window-scoped GETs that let the initial client load render each source independently (CONTEXT §5 "each source loads independently — events render as soon as events fetch completes, tasks when tasks fetch completes, etc. No waiting for slowest").

Purpose: SC1 (calendar renders three overlaid sources) + SC6 (live updates via SSE). The unified `/api/calendar/items` is retained as a convenience aggregator (used by the SSE poller, and still available for any non-hot code path that prefers one fetch). The per-source endpoints (`/api/actionables/with-due-dates`, `/api/personal-calendar/events` window-scoped, `/api/linkedin/posts/scheduled`) are what the dashboard's initial-load uses so each skeleton resolves on its own timeline.

Output: One new Fastify plugin registering the unified REST + SSE pair AND three per-source window-scoped GETs, two new query-layer helpers (actionables + events — pm-authority is called via the existing LinkedIn proxy client), the normalization logic that produces a single CalendarItem[] discriminated union, one vitest file. No changes to the dashboard yet — Plan 44-04 consumes these endpoints.
</objective>

## Discretion Decisions

**Initial-load topology — Option A (hybrid):** Keep the unified `/api/calendar/items` aggregator for the SSE poller (it already multiplexes the 3 sources into one hash-able envelope, which is the right shape for change detection). For the client's initial REST load, ALSO expose three per-source GETs so the dashboard can fire them in parallel and render each source's items as soon as its own fetch resolves — honoring CONTEXT §5's "no waiting for slowest" rule with the smallest diff. (Alternative B — emitting a per-source SSE stream — was rejected as heavier: more reconnect churn, more Zod schema duplication, and SSE live-updates are low-volume anyway.)

**Route placement:** The three per-source GETs live in the new calendar plugin (`src/api/routes/calendar.ts`) for cohesion with the aggregator + the window-clamp helper — NOT scattered across the existing actionables/personalCalendar/linkedin plugins. One `registerCalendarRoutes` call in `src/api/server.ts`, four routes, shared window-clamp constants. Tradeoff: route path sits on a plugin whose filename doesn't match 1-to-1 (e.g. `/api/actionables/with-due-dates` registered inside `calendar.ts`). Acceptable — the calendar plugin is named for its responsibility (unified calendar surface), not per the url prefix.

    ```ts
    /**
     * Calendar view: approved + fired actionables with a non-null fireAt
     * inside the window. Untimed approved actionables are excluded from
     * the calendar (they have nowhere to be placed). Phase 44 SC1.
     */
    export function getCalendarActionables(
      fromMs: number,
      toMs: number,
    ): Actionable[] {
      return db
        .select()
        .from(actionables)
        .where(
          and(
            inArray(actionables.status, ['approved', 'fired']),
            // fireAt IS NOT NULL AND BETWEEN fromMs AND toMs
            isNotNull(actionables.fireAt),
            gte(actionables.fireAt, fromMs),
            lte(actionables.fireAt, toMs),
          ),
        )
        .all();
    }
    ```

    ```ts
    import { createHash } from 'node:crypto';
    import { getCalendarActionables, type Actionable } from '../../db/queries/actionables.js';
    import { getApprovedEventsBetween } from '../../db/queries/personalPendingEvents.js';
    import { callUpstream } from '../linkedin/client.js';
    import { PostSchema } from '../linkedin/schemas.js';
    import { z } from 'zod';

    const POLL_INTERVAL_MS = 3_000;
    const HEARTBEAT_INTERVAL_MS = 15_000;
    const DEFAULT_FROM_OFFSET_MS = 7 * 24 * 60 * 60 * 1000;
    const DEFAULT_TO_OFFSET_MS = 60 * 24 * 60 * 60 * 1000;
    const MAX_WINDOW_MS = 120 * 24 * 60 * 60 * 1000;

    export type CalendarSource = 'task' | 'event' | 'linkedin';
    export type CalendarItem = {
      source: CalendarSource;
      id: string;
      title: string;
      start: number;            // unix ms
      end: number | null;       // unix ms, null for point-in-time
      isAllDay: boolean;
      language: 'he' | 'en' | 'mixed';
      sourceFields: Record<string, unknown>; // source-specific payload for the client
    };
    export type SourceStatus = 'ok' | 'error';
    export type CalendarEnvelope = {
      items: CalendarItem[];
      sources: { tasks: SourceStatus; events: SourceStatus; linkedin: SourceStatus };
    };
    ```

    Aggregator (private helper):

    ```ts
    async function fetchCalendarWindow(fromMs: number, toMs: number): Promise<CalendarEnvelope> {
      const [tasksRes, eventsRes, linkedinRes] = await Promise.allSettled([
        Promise.resolve().then(() => getCalendarActionables(fromMs, toMs)),
        Promise.resolve().then(() => getApprovedEventsBetween(fromMs, toMs)),
        callUpstream({
          method: 'GET',
          path: `/v1/posts?status=APPROVED`,
          timeoutMs: 5_000,
          responseSchema: z.array(PostSchema),
        }).then(({ data }) => data),
      ]);

      const items: CalendarItem[] = [];
      const sources: CalendarEnvelope['sources'] = { tasks: 'ok', events: 'ok', linkedin: 'ok' };

      if (tasksRes.status === 'fulfilled') {
        for (const a of tasksRes.value) {
          items.push({
            source: 'task',
            id: a.id,
            title: a.enrichedTitle ?? a.task,
            start: a.fireAt!,   // getCalendarActionables already filters NOT NULL
            end: null,
            isAllDay: false,
            language: a.detectedLanguage,
            sourceFields: {
              status: a.status,
              todoTaskId: a.todoTaskId,
              enrichedNote: a.enrichedNote,
              sourceContactName: a.sourceContactName,
            },
          });
        }
      } else {
        sources.tasks = 'error';
      }

      if (eventsRes.status === 'fulfilled') {
        for (const e of eventsRes.value) {
          items.push({
            source: 'event',
            id: e.id,
            title: e.title,
            start: e.eventDate,
            end: e.isAllDay ? null : e.eventDate + 60 * 60 * 1000, // default 1h span
            isAllDay: !!e.isAllDay,
            language: /[\u0590-\u05FF]/.test(e.title) ? 'he' : 'en',
            sourceFields: {
              location: e.location,
              description: e.description,
              calendarEventId: e.calendarEventId ?? null,
            },
          });
        }
      } else {
        sources.events = 'error';
      }

      if (linkedinRes.status === 'fulfilled') {
        for (const p of linkedinRes.value) {
          if (!p.scheduled_at) continue; // approved posts without a scheduled_at are unlikely but skip defensively
          const scheduledMs = new Date(p.scheduled_at).getTime();
          if (scheduledMs < fromMs || scheduledMs > toMs) continue;
          items.push({
            source: 'linkedin',
            id: p.id,
            title: p.content_he ?? p.content,   // Hebrew-first per CONTEXT §1
            start: scheduledMs,
            end: null,
            isAllDay: false,
            language: p.content_he ? 'he' : 'en',
            sourceFields: {
              status: p.status,
              content: p.content,
              content_he: p.content_he ?? null,
              image_urn: p.image_urn ?? null,
            },
          });
        }
      } else {
        sources.linkedin = 'error';
      }

      // Sort by start asc so the client can binary-search into the day grid.
      items.sort((a, b) => a.start - b.start);

      return { items, sources };
    }

    export function hashCalendarEnvelope(env: CalendarEnvelope): string {
      const compact = env.items.map(i => [i.source, i.id, i.title, i.start, i.end, i.isAllDay ? 1 : 0]);
      const statusBits = `${env.sources.tasks}:${env.sources.events}:${env.sources.linkedin}`;
      return createHash('sha1').update(JSON.stringify([compact, statusBits])).digest('hex');
    }
    ```

    ```ts
    // Shared query-parser so the unified route + the three per-source routes
    // apply the same window defaults + clamp.
    function parseWindow(q: { from?: string; to?: string }): { fromMs: number; toMs: number } {
      const now = Date.now();
      let fromMs = parseInt(q.from ?? '', 10);
      let toMs = parseInt(q.to ?? '', 10);
      if (!Number.isFinite(fromMs)) fromMs = now - DEFAULT_FROM_OFFSET_MS;
      if (!Number.isFinite(toMs)) toMs = now + DEFAULT_TO_OFFSET_MS;
      if (toMs - fromMs > MAX_WINDOW_MS) toMs = fromMs + MAX_WINDOW_MS;
      return { fromMs, toMs };
    }

    // Per-source projection helpers — extract the three blocks inside
    // fetchCalendarWindow into tiny private fns (projectTasks, projectEvents,
    // projectLinkedin) so they can be reused by the per-source routes. The
    // aggregator still calls them via Promise.allSettled.

    export default async function calendarRoutes(fastify: FastifyInstance) {
      // GET /api/calendar/items?from=<ms>&to=<ms>  (unified — used by SSE poller + any non-hot code path)
      fastify.get<{ Querystring: { from?: string; to?: string } }>(
        '/api/calendar/items',
        { onRequest: [fastify.authenticate] },
        async (request) => {
          const { fromMs, toMs } = parseWindow(request.query);
          return fetchCalendarWindow(fromMs, toMs);
        },
      );

      // Per-source GET #1: tasks (approved + fired actionables with fireAt in window)
      fastify.get<{ Querystring: { from?: string; to?: string } }>(
        '/api/actionables/with-due-dates',
        { onRequest: [fastify.authenticate] },
        async (request) => {
          const { fromMs, toMs } = parseWindow(request.query);
          const rows = getCalendarActionables(fromMs, toMs);
          return { items: projectTasks(rows) };
        },
      );

      // Per-source GET #2: events (approved personal events with eventDate in window)
      // NOTE: path is /api/personal-calendar/events with a ?from&to query-string
      // so the window-scoped read sits beside the existing list endpoint. If
      // the existing list route already occupies that path with different
      // semantics, register as /api/personal-calendar/events/window instead
      // and document the final path in the SUMMARY.
      fastify.get<{ Querystring: { from?: string; to?: string } }>(
        '/api/personal-calendar/events',
        { onRequest: [fastify.authenticate] },
        async (request) => {
          const { fromMs, toMs } = parseWindow(request.query);
          const rows = getApprovedEventsBetween(fromMs, toMs);
          return { items: projectEvents(rows) };
        },
      );

      // Per-source GET #3: LinkedIn (approved scheduled posts in window)
      fastify.get<{ Querystring: { from?: string; to?: string } }>(
        '/api/linkedin/posts/scheduled',
        { onRequest: [fastify.authenticate] },
        async (request, reply) => {
          const { fromMs, toMs } = parseWindow(request.query);
          try {
            const { data } = await callUpstream({
              method: 'GET',
              path: `/v1/posts?status=APPROVED`,
              timeoutMs: 5_000,
              responseSchema: z.array(PostSchema),
            });
            return { items: projectLinkedin(data, fromMs, toMs) };
          } catch (err) {
            return mapUpstreamErrorToReply(err, reply);
          }
        },
      );

      // GET /api/calendar/stream (SSE)
      fastify.get('/api/calendar/stream', async (request, reply) => {
        const { token } = request.query as { token?: string };
        try {
          fastify.jwt.verify(token ?? '');
        } catch {
          return reply.status(401).send({ error: 'Unauthorized' });
        }

        reply.raw.setHeader('Content-Type', 'text/event-stream');
        reply.raw.setHeader('Cache-Control', 'no-cache');
        reply.raw.setHeader('Connection', 'keep-alive');
        reply.raw.setHeader('X-Accel-Buffering', 'no');
        reply.raw.flushHeaders();

        let lastHash: string | null = null;
        let closed = false;
        const now0 = Date.now();
        // Stream window is same as REST default — 67-day span around now.
        const fromMs = now0 - DEFAULT_FROM_OFFSET_MS;
        const toMs = now0 + DEFAULT_TO_OFFSET_MS;

        const write = (frame: string) => {
          if (closed) return;
          try { reply.raw.write(frame); } catch { closed = true; }
        };

        const pollOnce = async () => {
          if (closed) return;
          try {
            const env = await fetchCalendarWindow(fromMs, toMs);
            const h = hashCalendarEnvelope(env);
            if (h !== lastHash) {
              lastHash = h;
              write(`event: calendar.updated\ndata: ${JSON.stringify(env)}\n\n`);
            }
          } catch (err) {
            fastify.log.warn({ err }, '[calendar-stream] poll failed; will retry');
          }
        };

        await pollOnce();
        const poll = setInterval(() => { void pollOnce(); }, POLL_INTERVAL_MS);
        const hb = setInterval(() => { write(': ping\n\n'); }, HEARTBEAT_INTERVAL_MS);

        request.raw.on('close', () => {
          closed = true;
          clearInterval(poll);
          clearInterval(hb);
        });
      });
    }
    ```
