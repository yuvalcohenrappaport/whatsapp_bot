    **Step C — `src/todo/todoService.ts`:** Append a new export `updateTodoTask` next to `createTodoTask` + `deleteTodoTask`. Read the existing file first to mirror its auth + error-handling shape (uses `google.tasks({version:'v1', auth})` + try/catch + `logger.error`). Mirror shape:

    ```ts
    export async function updateTodoTask(
      listId: string,
      taskId: string,
      patch: { title?: string; due?: string | null; notes?: string },
    ): Promise<boolean> {
      const auth = await getGoogleAuth(); // use same auth fn the existing helpers use
      if (!auth) {
        logger.warn('[todoService] no Google auth available for updateTodoTask');
        return false;
      }
      const tasks = getTasksApi(auth);
      try {
        await tasks.tasks.patch({
          tasklist: listId,
          task: taskId,
          requestBody: {
            ...(patch.title !== undefined && { title: patch.title }),
            ...(patch.due !== undefined && { due: patch.due }),
            ...(patch.notes !== undefined && { notes: patch.notes }),
          },
        });
        return true;
      } catch (err) {
        logger.error({ err }, '[todoService] updateTodoTask failed');
        return false;
      }
    }
    ```

    `due` takes an ISO8601 string or null to clear. The Google Tasks API requires ISO8601 for `due`. If the caller passes a unix ms, the route handler converts it to ISO first.

    ```ts
    export async function updatePersonalCalendarEvent(
      calendarId: string,
      eventId: string,
      patch: {
        summary?: string;
        description?: string;
        location?: string;
        start?: { dateTime?: string; date?: string; timeZone?: string };
        end?: { dateTime?: string; date?: string; timeZone?: string };
      },
    ): Promise<boolean> {
      const auth = await getGoogleAuth();
      if (!auth) {
        logger.warn('[personalCalendarService] no Google auth for update');

          calendarId,
          eventId,
          requestBody: patch,
        });
        return true;
      } catch (err) {
        logger.error({ err }, '[personalCalendarService] update event failed');
        return false;
      }
    }
    ```
