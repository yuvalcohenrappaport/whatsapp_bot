    ```ts
    import { useMutation, useQueryClient } from '@tanstack/react-query';
    import { toast } from 'sonner';
    import { apiFetch } from '@/api/client';
    import type { CalendarItem } from '@/api/calendarSchemas';

    type RescheduleArgs = { item: CalendarItem; toMs: number };

    export function useRescheduleMutation(opts?: {
      onOptimistic?: (item: CalendarItem, toMs: number) => void;
      onRollback?: (item: CalendarItem, fromMs: number) => void;
    }) {
      const qc = useQueryClient();
      return useMutation({
        mutationFn: async ({ item, toMs }: RescheduleArgs) => {
          if (item.source === 'task') {
            return apiFetch(`/api/actionables/${encodeURIComponent(item.id)}`, {
              method: 'PATCH',
              body: JSON.stringify({ fireAt: toMs }),
              headers: { 'Content-Type': 'application/json' },
            });
          }
          if (item.source === 'event') {
            return apiFetch(`/api/personal-calendar/events/${encodeURIComponent(item.id)}`, {
              method: 'PATCH',
              body: JSON.stringify({ eventDate: toMs }),
              headers: { 'Content-Type': 'application/json' },
            });
          }
          // linkedin
          return apiFetch(`/api/linkedin/posts/${encodeURIComponent(item.id)}/reschedule`, {
            method: 'POST',
            body: JSON.stringify({ scheduled_at: new Date(toMs).toISOString() }),
            headers: { 'Content-Type': 'application/json' },
          });
        },
        onMutate: async ({ item, toMs }) => {
          opts?.onOptimistic?.(item, toMs);
          return { fromMs: item.start };
        },
        onError: (err, { item }, ctx) => {
          if (ctx) opts?.onRollback?.(item, ctx.fromMs);
          toast.error(`Couldn't reschedule: ${err instanceof Error ? err.message : 'unknown'}`);
        },
        onSuccess: (response, { item, toMs }) => {
          // LinkedIn reschedule returns the PostDTO with the snapped scheduled_at.
          // If server-side snap moved the time, display the caption.
          let snappedMs = toMs;
          if (item.source === 'linkedin' && response && typeof response === 'object' && 'scheduled_at' in response) {
            const serverMs = new Date((response as { scheduled_at: string }).scheduled_at).getTime();
            if (Math.abs(serverMs - toMs) > 60_000) {
              snappedMs = serverMs;
              toast(`Moved to next available slot: ${new Date(serverMs).toLocaleString('en-GB', { timeZone: 'Asia/Jerusalem' })}`);
              opts?.onOptimistic?.(item, serverMs);
            }
          }
          // Undo toast (5s)
          toast(`Moved "${item.title.slice(0, 40)}"`, {
            duration: 5000,
            action: {
              label: 'Undo',
              onClick: () => {
                // Trigger a reverse mutation inline.
                void (async () => {
                  try {
                    if (item.source === 'task') {
                      await apiFetch(`/api/actionables/${encodeURIComponent(item.id)}`, {
                        method: 'PATCH',
                        body: JSON.stringify({ fireAt: item.start }),
                        headers: { 'Content-Type': 'application/json' },
                      });
                    } else if (item.source === 'event') {
                      await apiFetch(`/api/personal-calendar/events/${encodeURIComponent(item.id)}`, {
                        method: 'PATCH',
                        body: JSON.stringify({ eventDate: item.start }),
                        headers: { 'Content-Type': 'application/json' },
                      });
                    } else {
                      await apiFetch(`/api/linkedin/posts/${encodeURIComponent(item.id)}/reschedule`, {
                        method: 'POST',
                        body: JSON.stringify({ scheduled_at: new Date(item.start).toISOString() }),
                        headers: { 'Content-Type': 'application/json' },
                      });
                    }
                    opts?.onOptimistic?.(item, item.start);
                    toast(`Undone`);
                  } catch (err) {
                    toast.error('Undo failed');
                  }
                })();
              },
            },
          });
          void snappedMs;
        },
      });
    }

    // Similar shape: useInlineEditMutation({item, newTitle}) — routes to
    // PATCH /api/actionables/:id {task} for task, PATCH /api/personal-calendar/events/:id {title} for event,
    // POST /api/linkedin/posts/:id/edit {content: newTitle, content_he: newTitle if language==='he'} for linkedin.
    // Optimistic + error toast. No undo toast (inline edits are lower-risk).

    // useCreateMutation({source, fields}) — routes to POST /api/actionables,
    // POST /api/personal-calendar/events, or the appropriate LinkedIn endpoint
    // (LinkedIn create-from-calendar is limited in v1 to forwarding to the full
    // pm-authority lesson-run flow — for the MVP, a LinkedIn create popover
    // OPENS the LessonGenerationModal or navigates to /linkedin/queue instead
    // of a direct POST, since pm-authority has no public "create post with
    // content + scheduled_at" endpoint. Task + event creates POST directly.)
    ```
