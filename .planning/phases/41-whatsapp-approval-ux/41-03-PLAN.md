```typescript
export async function tryHandleApprovalReply(
  sock: WASocket,
  text: string,
  quotedMsgId: string,
  detectedLanguage: 'he' | 'en', // of the reply text (use calendarApproval.detectMessageLanguage on the caller side)
): Promise<boolean> {
  // 1. Find all actionables whose approval_preview_message_id === quotedMsgId,
  //    ordered by created_at asc so item indices are stable with the preview.
  const items = getActionablesByPreviewMsgId(quotedMsgId); // new query fn — see Task 2
  if (items.length === 0) return false;

  // 2. Parse the reply using the item count
  const directives = parseApprovalReply(text, items.length);

  // 3. If empty → grammar-hint reply, return true (handled)
  if (directives.length === 0) {
    await sendGrammarHint(sock, items[0].detectedLanguage as 'he' | 'en');
    return true;
  }

  // 4. Expand 'all' directives into one per item
  const expanded = expandAllDirectives(directives, items.length);

  // 5. Apply each directive in input order (de-dupe by itemIndex keeping last)
  for (const dir of dedupLast(expanded)) {
    const actionable = items[dir.itemIndex - 1];
    await applyDirective(sock, actionable, dir);
  }
  return true;
}

async function applyDirective(sock, actionable, directive) {
  // Skip + warn if already-handled
  if (actionable.status !== 'pending_approval') {
    await sock.sendMessage(config.USER_JID, { text: alreadyHandledMsg(directive, actionable) });
    return;
  }

  if (directive.action === 'edit') {
    updateActionableTask(actionable.id, directive.editText!);
    actionable.task = directive.editText!; // local refresh for the approve path below
  }

  if (directive.action === 'approve' || directive.action === 'edit') {
    // Push to Google Tasks using raw task text (Phase 42 will intercept to enrich)
    await approveAndSync(sock, actionable);
    return;
  }

  if (directive.action === 'reject') {
    updateActionableStatus(actionable.id, 'rejected');
    await sock.sendMessage(config.USER_JID, { text: rejectedConfirmation(actionable.detectedLanguage) });
    return;
  }
}

async function approveAndSync(sock, actionable) {
  updateActionableStatus(actionable.id, 'approved');
  if (isTasksConnected()) {
    try {
      const result = await createTodoTask({
        title: actionable.task, // raw — enrichment is Phase 42
        note: buildBasicNote(actionable),
      });
      updateActionableTodoIds(actionable.id, result);
    } catch (err) {
      logger.warn({ err, id: actionable.id }, 'Google Tasks push failed on approval');
    }
  }
  await sock.sendMessage(config.USER_JID, { text: approvedConfirmation(actionable) });
}

function buildBasicNote(a): string {
  const who = a.sourceContactName ?? a.sourceContactJid ?? 'Self';
  const snippet = a.sourceMessageText ? `\nOriginal: "${truncate(a.sourceMessageText, 200)}"` : '';
  return `From: ${who}${snippet}`;
}

function approvedConfirmation(a): string {
  if (a.detectedLanguage === 'he') return `✅ נוסף: ${a.task}`;
  return `✅ Added: ${a.task}`;
}

function rejectedConfirmation(lang: 'he' | 'en'): string {
  return lang === 'he' ? '❌ בוטל' : '❌ Dismissed';
}

function alreadyHandledMsg(directive, actionable): string {
  if (actionable.detectedLanguage === 'he') {
    return `⚠️ פריט ${directive.itemIndex} כבר טופל (${actionable.status}) — דולג.`;
  }
  return `⚠️ Item ${directive.itemIndex} already handled (${actionable.status}) — skipped.`;
}

async function sendGrammarHint(sock, lang): Promise<void> {
  // Single-line hint matching the preview's trailing line
  const text = lang === 'he'
    ? "השב ✅ / ❌ / עריכה: <טקסט> (או מספר + פעולה עבור פריט ספציפי)"
    : "Reply ✅ / ❌ / edit: <text> (or number + action for a specific item)";
  await sock.sendMessage(config.USER_JID, { text });
}
```
