```typescript
const DEBOUNCE_MS = 2 * 60 * 1000; // 2 minutes per 40-CONTEXT.md

interface Bucket {
  actionableIds: string[];
  timer: NodeJS.Timeout;
  sourceContactJid: string;
}

const buckets = new Map<string, Bucket>();

export type FlushCallback = (actionableIds: string[]) => void | Promise<void>;

let flushCallback: FlushCallback | null = null;
export function setFlushCallback(cb: FlushCallback): void {
  flushCallback = cb;
}

export function enqueueForPreview(
  actionableId: string,
  sourceContactJid: string,
): void {
  const existing = buckets.get(sourceContactJid);
  if (existing) {
    clearTimeout(existing.timer);
    existing.actionableIds.push(actionableId);
    existing.timer = setTimeout(() => flush(sourceContactJid), DEBOUNCE_MS);
    return;
  }
  const bucket: Bucket = {
    actionableIds: [actionableId],
    sourceContactJid,
    timer: setTimeout(() => flush(sourceContactJid), DEBOUNCE_MS),
  };
  buckets.set(sourceContactJid, bucket);
}

async function flush(sourceContactJid: string): Promise<void> {
  const bucket = buckets.get(sourceContactJid);
  if (!bucket) return;
  buckets.delete(sourceContactJid);
  if (!flushCallback) return;
  try {
    await flushCallback(bucket.actionableIds);
  } catch (err) {
    // Log and swallow — a failed flush should not crash the bot
  }
}

/** Test-only: clear all buckets and their timers. */
export function __resetBucketsForTest(): void {
  for (const bucket of buckets.values()) clearTimeout(bucket.timer);
  buckets.clear();
  flushCallback = null;
}

/** Test-only: inspect bucket state. */
export function __getBucketForTest(jid: string): Bucket | undefined {
  return buckets.get(jid);
}
```

The flush is fire-and-forget from the timer perspective; setFlushCallback decouples the bucket from the sender so unit tests can mock the callback.

`setFlushCallback` is called once at bot init (wire this up in Phase 41 Plan 41-04 or via a new init helper — for this plan, just export it and let downstream plans register).

import pino from 'pino';
import { config } from '../config.js';
import { getState } from '../api/state.js';
import {
  getActionableById,
  updateActionablePreviewMsgId,
} from '../db/queries/actionables.js';
import { composePreview } from './previewTemplates.js';

const logger = pino({ level: config.LOG_LEVEL });

export async function sendBucketPreview(
  actionableIds: string[],
): Promise<void> {
  if (actionableIds.length === 0) return;
  const items = actionableIds
    .map((id) => getActionableById(id))
    .filter((a): a is NonNullable<typeof a> => !!a)
    .filter((a) => a.status === 'pending_approval'); // drop already-acted items

  if (items.length === 0) return;

  // All items in a bucket come from the same sourceContactJid by construction,
  // so language + contactName are consistent.
  const language = items[0].detectedLanguage as 'he' | 'en';
  const contactName = items[0].sourceContactName ?? null;

  const previewItems = items.map((a) => ({
    task: a.task,
    contactName: a.sourceContactName,
    snippet: truncate(a.sourceMessageText, 100),
  }));

  const messageText = composePreview(previewItems, language, contactName);

  const sock = getState().sock;
  if (!sock) {
    logger.warn({ actionableIds }, 'No sock — cannot send preview');
    return;
  }

  try {
    const sent = await sock.sendMessage(config.USER_JID, {
      text: messageText,
    });
    const previewMsgId = sent?.key?.id;
    if (!previewMsgId) {
      logger.error({ actionableIds }, 'sendMessage returned no key.id');
      return;
    }
    // Annotate every bucketed actionable with the preview msg id so the
    // quoted-reply matcher in Plan 41-03 can find them.
    for (const a of items) {
      updateActionablePreviewMsgId(a.id, previewMsgId);
    }
    logger.info(
      { previewMsgId, count: items.length, language, contactName },
      'Sent actionables preview',
    );
  } catch (err) {
    logger.error({ err, actionableIds }, 'Failed to send actionables preview');
  }
}

function truncate(s: string, max: number): string {
  if (!s) return '';
  if (s.length <= max) return s;
  return s.slice(0, max).trimEnd() + '…';
}
